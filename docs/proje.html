1 Gerekli paketler

install.packages(c(“ggplot2”, “patchwork”, “grid”, “scales”, “dplyr”))

library(ggplot2) library(patchwork) library(grid) library(scales) library(dplyr)

2 Veri seti

trafik <- data.frame( yil = 2009:2024, nufus = c(72561312,73722988,74724269,75627384,76667864,77695904,78741053,79814871, 80810525,82003882,83154997,83614362,84680273,85279553,85372377,85664944), arac_sayisi = c(14316700,15095603,16089528,17033413,17939447,18828721,19994472, 21090424,22218945,22865921,23156975,24144857,25249119,26482847, 28740492,31301389), kaza_sayisi = c(1053346,1106201,1228928,1296634,1207354,1199010,1313359,1182491, 1202716,1229364,1168144,983808,1186353,1232957,1314136,1444026) )

3 Türetilmiş değişkenler

trafik <- trafik %>% mutate( kaza_10000_arac = kaza_sayisi / arac_sayisi * 10000, nufus_endeks = nufus / first(nufus) * 100, arac_endeks = arac_sayisi / first(arac_sayisi) * 100, kaza_endeks = kaza_sayisi / first(kaza_sayisi) * 100 )

4 Tema

tema <- theme_minimal(base_size = 11) + theme( plot.title = element_text(face = “bold”, size = 13, color = “#0F172A”), axis.title = element_text(size = 10), axis.text.x = element_text(angle = 45, hjust = 1), panel.grid.minor = element_blank() )

5 Grafik 1

g1 <- ggplot(trafik, aes(yil, arac_sayisi)) + geom_line(linewidth = 1.1, color = “#2563EB”) + geom_point(size = 2, color = “#2563EB”) + scale_y_continuous(labels = label_number(big.mark = “.”)) + scale_x_continuous(breaks = trafik$yil) + labs(title = “Yıllara Göre Araç Sayısı”, x = “Yıl”, y = “Araç Sayısı”) + tema

6 Grafik 2

g2 <- ggplot(trafik, aes(yil, kaza_sayisi)) + geom_line(linewidth = 1.1, color = “#DC2626”) + geom_point(size = 2, color = “#DC2626”) + scale_y_continuous(labels = label_number(big.mark = “.”)) + scale_x_continuous(breaks = trafik$yil) + labs(title = “Yıllara Göre Toplam Kaza Sayısı”, x = “Yıl”, y = “Kaza Sayısı”) + tema

7 Grafik 3 - Endeks

endeks_long <- trafik %>% select(yil, nufus_endeks, arac_endeks, kaza_endeks) %>% tidyr::pivot_longer( cols = -yil, names_to = “degisken”, values_to = “endeks” ) %>% mutate( degisken = recode( degisken, “nufus_endeks” = “Nüfus”, “arac_endeks” = “Araç Sayısı”, “kaza_endeks” = “Kaza Sayısı” ) )

g3 <- ggplot(endeks_long, aes(yil, endeks, color = degisken)) + geom_line(linewidth = 1.2) + geom_point(size = 2) + scale_x_continuous(breaks = trafik$yil) + labs( title = “2009=100 Endeksine Göre Değişim”, x = “Yıl”, y = “Endeks Değeri”, color = “Değişken” ) + tema + theme(legend.position = “bottom”)

8 Grafik 4

g4 <- ggplot(trafik, aes(yil, kaza_10000_arac)) + geom_line(linewidth = 1.1, color = “#16A34A”) + geom_point(size = 2, color = “#16A34A”) + scale_x_continuous(breaks = trafik$yil) + labs(title = “10.000 Araç Başına Kaza Sayısı”, x = “Yıl”, y = “Kaza Sayısı”) + tema
